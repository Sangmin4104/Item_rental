CREATE TABLE ID(
SNUM INT NOT NULL PRIMARY KEY,
DNAME VARCHAR(50),
SNAME VARCHAR(20),
SPHONE INT,
SHEAD INT
);


CREATE TABLE PROD(
P_PNUM INT NOT NULL PRIMARY KEY,
P_PNAME VARCHAR(30),
P_AMOUNT INT,
P_RENTAL VARCHAR(30),
P_SNUM INT,
P_IMAGES VARCHAR(30),
P_STATE VARCHAR(30), 
CONSTRAINT fk_code1 FOREIGN KEY (P_SNUM) REFERENCES ID (SNUM)
);

CREATE TABLE PROD_DOC(
PR_YID INT NOT NULL PRIMARY KEY,
PR_PNUM INT,
PR_SNUM INT,
PR_AMOUNT INT,
PR_START DATE,
PR_END DATE,
PR_CANCEL VARCHAR(50),
CONSTRAINT fk_code2 FOREIGN KEY (PR_PNUM) REFERENCES PROD (P_PNUM),
CONSTRAINT fk_code3 FOREIGN KEY (PR_SNUM) REFERENCES ID (SNUM)
);

CREATE TABLE C_CHECK(
C_YID INT,
C_SNUM INT,
C_PNUM INT,
C_CHECK VARCHAR(50),
C_START DATE,
C_CDATE DATE,
CONSTRAINT fk_code4 FOREIGN KEY (C_YID) REFERENCES PROD_DOC (PR_YID),
CONSTRAINT fk_code5 FOREIGN KEY (C_SNUM) REFERENCES ID (SNUM),
CONSTRAINT fk_code6 FOREIGN KEY (C_PNUM) REFERENCES PROD (P_PNUM)
);
